'use strict';
 
function AutocompleteDirective($locale) {
 
    return {
        restrict: 'A',
        require: '?ngModel',
        link: function(scope, element, attrs, ctrl) {
 
            function loadData(arr, key, lim) {
 
                if (!lim) lim = 10;
                return function(txt, fnc) {
 
                    var result, f, filter;
                    result = [],
                        filter = new RegExp(txt, "i"),
                        $.each(arr,
                            function(arr, obj) {
 
                                (
                                    (key && filter.test(obj[key])) ||
                                    (!key && filter.test(JSON.stringify(obj)))
                                ) && result.length < lim && result.push({
                                    obj: obj,
                                    name: obj.name
                                })
                            }), fnc(result)
                }
            }
 
            attrs.$observe('fluigAutocomplete', function() {
 
                var dataset = attrs.fluigAutocomplete;
 
                if (dataset == '' || !dataset) return;
 
                scope.$watch(dataset, function(value) {
                    if (!value) return;
                    var ac = FLUIGC.autocomplete(element, {
                            source: loadData(value, 'name'),
                            minLength: 0,
                            highlight: true,
                            displayKey: 'name',
                            maxTags: 1,
                            type: 'autocomplete'
                        })
                        .on('fluig.autocomplete.selected', function(result) {
                            ctrl.$setViewValue(result.item.obj);
                        })
                        .on('fluig.autocomplete.itemRemoved', function(result) {
                            ctrl.$setViewValue();
                        });
 
                    if (ctrl.$modelValue) {
                        ac.val(ctrl.$modelValue.name);
                    }
 
                });
            });
 
        }
    };
}
 
module.exports = AutocompleteDirective;
